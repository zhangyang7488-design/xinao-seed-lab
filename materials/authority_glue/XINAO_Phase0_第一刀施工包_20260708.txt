## 参考代码骨架（Codex 可展开）

```python
from temporalio import activity
from markitdown import MarkItDown
from e2b_code_interpreter import Sandbox
import git, json
from datetime import datetime
from pathlib import Path

EVIDENCE_DIR = Path(r"D:\XINAO_RESEARCH_RUNTIME\readback")
S_WORKSPACE = Path(r"E:\XINAO_RESEARCH_WORKSPACES\S")

@activity.defn
async def phase0_minimal_intake_and_execute(test_input_path: str) -> dict:
    md = MarkItDown()
    task_md = md.convert(test_input_path).text_content
    task_package = {"source": test_input_path, "content_md": task_md[:2000],
                    "timestamp": datetime.now().isoformat()}
    with Sandbox.create() as sandbox:
        exec_output = sandbox.run_code(
            f'print("Phase0 {datetime.now()}")\nprint("{task_md[:300]}...")'
        ).text
    repo = git.Repo(S_WORKSPACE)
    proof = S_WORKSPACE / "phase0_proof.txt"
    proof.write_text(f"{datetime.now()}\n{exec_output}\n")
    repo.index.add([str(proof)])
    commit = repo.index.commit("Phase0: minimal weld")
    EVIDENCE_DIR.mkdir(parents=True, exist_ok=True)
    evidence_file = EVIDENCE_DIR / f"phase0_{datetime.now().strftime('%Y%m%d_%H%M%S')}.json"
    evidence = {"phase": "0", "intake": task_package, "execution_output": exec_output,
                "commit_hash": commit.hexsha, "evidence_path": str(evidence_file)}
    evidence_file.write_text(json.dumps(evidence, ensure_ascii=False, indent=2))
    return evidence
```

运行：
  cd E:\XINAO_RESEARCH_WORKSPACES\S
  python -m services.agent_runtime.phase0_minimal_weld_activity
